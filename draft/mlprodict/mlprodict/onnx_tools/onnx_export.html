    def numpy_mlprodict_ONNX_KMeans(X):
        '''
        Numpy function for ``mlprodict_ONNX_KMeans``.
    
        * producer: skl2onnx
        * version: 0
        * description: 
        '''
        # initializers
    
        Ad_Addcst = numpy.array([421.0, 41.0], dtype=numpy.float32)
    
        Ge_Gemmcst = numpy.array([14.0, 15.0, 4.0, 5.0],
                                 dtype=numpy.float32).reshape((2, 2))
    
        Mu_Mulcst = numpy.array([0.0], dtype=numpy.float32)
    
        # nodes
    
        Re_reduced0 = (X ** 2).sum(axis=1, keepdims=1)
        Mu_C0 = Re_reduced0 * Mu_Mulcst
        Ge_Y0 = X @ Ge_Gemmcst.T * -2.0 + Mu_C0 * 0.0
        Ad_C01 = Re_reduced0 + Ge_Y0
        Ad_C0 = Ad_Addcst + Ad_C01
        scores = Ad_C0 ** 0.5
        label = argmin_use_numpy_select_last_index(
            Ad_C0, axis=1, keepdims=0, select_last_index=0)
    
        return label, scores
