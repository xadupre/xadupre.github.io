    import onnx

    node = onnx.helper.make_node(
        "Asinh",
        inputs=["x"],

    x = np.array([-1, 0, 1]).astype(np.float32)
    y = np.arcsinh(x)  # expected output [-0.88137358,  0.,  0.88137358]
